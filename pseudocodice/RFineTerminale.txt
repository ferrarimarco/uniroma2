* DESCRIZIONE PROCEDURA < Fine Terminale >;
PROCEDURE Fine Terminale;
BEGIN

	* GESTIONE JOB USCENTE DA TERMINALE
	
	< creo nuovo JOB J classe 1 >;
	< inizializzo J >;
	< creo nuovo HOST H per J se necessario >;

	* CREO STAMPANTE PER JOB
	< creo nuova STAMPANTE ST per J >;

	* MEMORIZZA IN CLASSE JOB ISTANTE DI USCITA JOB DA TERMINALE PER CALCOLO TEMPO MEDIO DI RISPOSTA
	< istante di uscita J da TERM := clock >;
	
	* INVIO A CPU
	< prossimo centro NX(J) = CPU per J >;

	IF < CPU libera >
		THEN
		< inserisco J in CPU >;
	ELSE
		< inserisco J in codaCpu >;

	* LIBERO TERMINALE
	< libera terminale associato a J >;

	* AGGIORNAMENTO CALENDARIO
	< prevedi durata Tcpu(J) del servizio cpu >;
	< aggiorno calendario tcpu := clock + Tcpu(J) >;

END
