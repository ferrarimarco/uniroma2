* DESCRIZIONE PROCEDURA  <FINE DISK>
PROCEDURE Fine Disk;
BEGIN

	* GESTIONE JOB USCENTE
	* INVIO A CPU
	< prossima unità NX(J) = CPU per J >;

	* DECIDO SE HO TROVATO RECORD
	IF < prob <= 0.1 >
		THEN
		< record trovato >;
	ELSE
		< record non trovato >;
	
	* USCITA JOB DA DISK
	
	* INCREMENTA CONTATORE JOB USCITI
	< jDisk++ >;	
	
	IF < CPU libera >
		THEN
		< inserisco J in CPU >;
	ELSE
		< inserisco J in codaCpuClasse3 >;

	* AGGIORNAMENTO CALENDARIO
	< prevedi Tcpu(J) servizio CPU >;
	< tcpu:= clock + Tcpu(J) >;
		
	* GESTIONE JOB ENTRANTE
	IF <codaDisk non vuota>
	   THEN 
		* MEMORIZZA ISTANTE DI ENTRATA JOB IN DISK PER CALCOLO THROUGHPUT
		< istante di entrata J in DISK := clock >;
				
		< occupo DISK con J >;
	ELSE
	   < libera Disk >;
END
